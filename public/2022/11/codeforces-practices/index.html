#include <stdio.h>
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int main()
{
    int t, n;
    scanf("%d", &t);
    while (t--)
    {
        scanf("%d", &n);
        if (n & 1)
        {
            for (int i = 1; i <= n; i++)
            {
                printf("%d",n);
                if(i!=n){
                    printf(" ");
                }
            }
            printf("\n");
            continue;
        } else {
            printf("1");
            for (int i = 1; i <= n-1; i++)
            {
                printf(" %d",n+1);
            }
            printf("\n");
        }
    }
    return 0;
}
